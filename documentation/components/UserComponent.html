import { Component, EventEmitter, Output, inject } from '@angular/core';
import { SessionState } from '../../core/state/session.state';

@Component({
  selector: 'app-user',
  templateUrl: './user.component.html',
})
export class UserComponent {
  public readonly session = inject(SessionState);

  @Output() productsAdmin = new EventEmitter<void>();
  @Output() categoriesAdmin = new EventEmitter<void>();
  @Output() orders = new EventEmitter<void>();
}

<section class="simple-screen profile-screen" aria-label="Perfil">
  <div class="profile-card">
    <img src="/icons/tdesign/user.svg" alt="" aria-hidden="true" />
    <h2>{{ session.currentTestUser.nombre }}</h2>
    <p>{{ session.isOwnerRole ? 'Modo de pruebas para administracion del catalogo.' : 'Modo de pruebas para experiencia de cliente.' }}</p>
  </div>

  <section class="profile-role-switch" aria-label="Cambiar rol de prueba">
    <div class="profile-switch-copy">
      <strong>Rol activo</strong>
      <small>{{ session.isOwnerRole ? 'Dueno' : 'Cliente' }}</small>
    </div>
    <div class="profile-switch-buttons">
      <button type="button" [class.active]="session.currentTestRole === 'dueno'" (click)="session.switchRole('dueno')">Dueno</button>
      <button type="button" [class.active]="session.currentTestRole === 'cliente'" (click)="session.switchRole('cliente')">Cliente</button>
    </div>
  </section>

  @if (session.isOwnerRole) {
    <button class="profile-action" type="button" (click)="productsAdmin.emit()">Productos</button>
    <button class="profile-secondary" type="button" (click)="categoriesAdmin.emit()">Categorias</button>
  }
  <button class="profile-secondary" type="button" (click)="orders.emit()">Mis Pedidos</button>
  <button class="profile-secondary" type="button">Direcciones</button>
</section>
